## 二、浏览器插件核心功能

### 功能1：捕捉 AI 对话上下文

**技术方案**：Content Script + MutationObserver

```javascript
// content-script.js
class ConversationCapture {
  constructor() {
    this.conversations = [];
    this.observer = null;
    this.adapters = {
      'chatgpt.com': new ChatGPTAdapter(),
      'claude.ai': new ClaudeAdapter(),
      'gemini.google.com': new GeminiAdapter(),
      // 支持更多网站
    };
  }

  init() {
    const hostname = window.location.hostname;
    const adapter = this.findAdapter(hostname);

    if (!adapter) {
      console.log('当前网站不支持对话捕捉');
      return;
    }

    // 监听 DOM 变化
    this.observer = new MutationObserver((mutations) => {
      for (const mutation of mutations) {
        if (mutation.addedNodes.length > 0) {
          this.detectNewMessages(mutation.addedNodes, adapter);
        }
      }
    });

    this.observer.observe(document.body, {
      childList: true,
      subtree: true
    });
  }

  detectNewMessages(nodes, adapter) {
    for (const node of nodes) {
      if (node.nodeType !== Node.ELEMENT_NODE) continue;

      // 使用适配器识别消息
      const message = adapter.extractMessage(node);
      if (message) {
        this.conversations.push(message);
        this.autoSave(message);
      }
    }
  }

  async autoSave(message) {
    // 调用本地知识库 API
    await fetch('http://localhost:8080/api/memories', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        text: message.content,
        category: message.role === 'assistant' ? 'fact' : 'preference',
        scope: `browser:${window.location.hostname}`,
        metadata: {
          source: 'browser-plugin',
          url: window.location.href,
          timestamp: message.timestamp,
          role: message.role
        }
      })
    });
  }
}

// 初始化
const capture = new ConversationCapture();
capture.init();
```

### 网站适配器示例

```javascript
// adapters/chatgpt.js
class ChatGPTAdapter {
  extractMessage(node) {
    // ChatGPT 的消息结构
    if (!node.matches('[data-message-author-role]')) return null;

    const role = node.getAttribute('data-message-author-role');
    const contentEl = node.querySelector('.markdown');

    if (!contentEl) return null;

    return {
      role: role, // 'user' or 'assistant'
      content: contentEl.innerText,
      timestamp: Date.now()
    };
  }
}

// adapters/claude.js
class ClaudeAdapter {
  extractMessage(node) {
    // Claude 的消息结构
    if (node.matches('[data-test-render-count]')) {
      const isUser = node.querySelector('.font-user-message') !== null;
      const content = node.querySelector('.font-claude-message, .font-user-message');

      if (!content) return null;

      return {
        role: isUser ? 'user' : 'assistant',
        content: content.innerText,
        timestamp: Date.now()
      };
    }
    return null;
  }
}
```

### 功能2：召回记忆按钮

**技术方案**：在输入框旁边注入按钮，点击后召回相关记忆

```javascript
// recall-button.js
class RecallButton {
  constructor() {
    this.button = null;
  }

  inject(inputElement) {
    // 创建按钮
    this.button = document.createElement('button');
    this.button.innerHTML = '🧠 召回记忆';
    this.button.className = 'memory-recall-btn';
    this.button.style.cssText = `
      position: absolute;
      right: 60px;
      bottom: 10px;
      padding: 6px 12px;
      background: #4CAF50;
      color: white;
      border: none;
      border-radius: 4px;
      cursor: pointer;
      z-index: 9999;
    `;

    // 点击事件
    this.button.addEventListener('click', async () => {
      await this.recallMemories(inputElement);
    });

    // 插入到输入框旁边
    const parent = inputElement.parentElement;
    parent.style.position = 'relative';
    parent.appendChild(this.button);
  }

  async recallMemories(inputElement) {
    const currentInput = inputElement.value;

    // 调用检索 API
    const response = await fetch(`http://localhost:8080/api/search?q=${encodeURIComponent(currentInput)}&limit=3`);
    const results = await response.json();

    if (results.length === 0) {
      alert('未找到相关记忆');
      return;
    }

    // 格式化记忆
    const memoryText = results.map((r, i) =>
      `[记忆${i+1}] ${r.memory.summary || r.memory.raw_text}`
    ).join('\n\n');

    // 插入到输入框
    const prefix = '以下是相关的历史记忆：\n\n';
    const suffix = '\n\n---\n我的问题：' + currentInput;
    inputElement.value = prefix + memoryText + suffix;

    // 触发输入事件（某些网站需要）
    inputElement.dispatchEvent(new Event('input', { bubbles: true }));
  }
}

// 初始化
const recallBtn = new RecallButton();
// 找到输入框并注入按钮
const inputBox = document.querySelector('textarea[placeholder*="Message"]');
if (inputBox) {
  recallBtn.inject(inputBox);
}
```
